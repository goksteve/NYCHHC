select a.DOS, a.DATEORDERED, 
   D.FACILITY_NAME, D.MRN, D.MEMBERID,
   a.MEMBERNUM, c.MEMBERNUM, c.LASTNAME, c.FIRSTNAME, c.SUBSNUM,
   a.CLAIMNUM, a.NDCNUM, 
   a.DRUGNAME, a.QUANTITY, a.DAYSSUPPLY, 
   a.REFILLNUM, a.PRESCRIBERID, b.PRESCRIBERID, b.PRESLASTNAME, b.PRESFIRSTNAME, a.DEANUM, 
   a.PHARMACYNUM, a.PHARMACYNAME, a.PHARMACYADDRESS, 
   a.PHARMACYCITY, a.PHARMACYSTATE, a.PHARMACYZIP

from STG_PHARMACY a

left join STG_PHA_PRESCRIBER b on b.PRESCRIBERID = a.PRESCRIBERID
join STG_MEMMAS_FINAL c on c.MEMBERNUM = a.MEMBERNUM
left join METROPLUS_ASSIGNED_MRN d ON d.MEMBERID = a.MEMBERNUM
